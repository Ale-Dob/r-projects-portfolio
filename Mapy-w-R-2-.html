  geom_sf(aes(geometry=srodki,
              size = LICZBA_MIE),
          fill= "deepskyblue",
          alpha = 0.5,
          shape = 22)+
  theme_void()+
  scale_size_continuous(
    name = "liczba mieszkacw",
    range = c(2,12), #zakres rozmiaru punktw
    breaks = c(2000, 5000, 10000, 20000),
    guide = guide_legend(reverse = T)
